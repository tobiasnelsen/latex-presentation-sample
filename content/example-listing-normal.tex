\begin{frame}{Normal listing example}

An \textit{itemize} listing:

\begin{itemize}
  \item First item
  \item Second item
\end{itemize}

\vspace{0.6cm}

An \textit{enumerate} listing:

\begin{enumerate}
  \item First item
  \item Second item
\end{enumerate}

\vspace{0.6cm}

An \textit{enumerate} listing with a different symbol:

\begin{enumerate}[1)]
  \item First item
  \item Second item
\end{enumerate}

\end{frame}
